document.getElementById('downloadScreenshot').addEventListener('click', function() { var element = document.body; var originalBackground = element.style.background; // Apply the background gradient explicitly to the element before screenshot element.style.background = 'linear-gradient(to bottom right, #fff8eb, #e2d8e6)'; var currentWidth = window.innerWidth; var currentHeight = element.scrollHeight; var scaleFactor = 4; // Clone the element to avoid modifying the original layout var clonedElement = element.cloneNode(true); clonedElement.style.width = currentWidth + 'px'; clonedElement.style.height = currentHeight + 'px'; clonedElement.style.transform = 'scale(' + scaleFactor + ')'; clonedElement.style.transformOrigin = 'top left'; clonedElement.style.position = 'absolute'; clonedElement.style.left = '-10000px'; // Position it off-screen document.body.appendChild(clonedElement); domtoimage.toPng(clonedElement, { width: currentWidth * scaleFactor, height: currentHeight * scaleFactor, }).then(function (dataUrl) { var link = document.createElement('a'); link.download = 'Lightful_share_image.png'; link.href = dataUrl; link.click(); // Remove the clone after the screenshot document.body.removeChild(clonedElement); element.style.background = originalBackground; }).catch(function (error) { console.error('Error capturing screenshot:', error); }); });
